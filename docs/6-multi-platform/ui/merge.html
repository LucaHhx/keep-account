# 生成全平台图标的命令

cd src-tauri

cargo tauri icon ../docs/6-multi-platform/ui/Resources/icons/icon-source-1024.png

# iOS / Android 需要先初始化

cargo tauri ios init

cargo tauri android init

cargo tauri icon ../docs/6-multi-platform/ui/Resources/icons/icon-source-1024.png
